

SELECT nome, salario

SELECT nome, data_contratacao
FROM funcionarios 
WHERE local_de_trabalho = "salao" AND data_contratacao > '2004-01-01';

SELECT nome, data_nascimento
FROM funcionarios 
WHERE nome LIKE "%ei%"

FROM funcionarios 
WHERE data_nascimento < '1998-01-01' AND salario > 1300
